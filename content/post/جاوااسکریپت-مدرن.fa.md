# جاوااسکریپت مدرن: از ES6 تا امروز

## متغیرها: let و const
```javascript
// قدیمی
var name = "Ali";

// مدرن
let age = 30;
const PI = 3.14;
توابع arrow
javascript
// قدیمی
function add(a, b) {
    return a + b;
}

// مدرن
const add = (a, b) => a + b;
Template Literals
javascript
const name = "Ali";
console.log(`سلام ${name}!`); // سلام Ali!
Destructuring
javascript
const person = { name: "Ali", age: 30 };
const { name, age } = person;
Modules
javascript
// export
export const PI = 3.14;
export function calculate() { ... }

// import
import { PI } from './math.js';
Async/Await
javascript
async function fetchData() {
    try {
        const response = await fetch(url);
        const data = await response.json();
        return data;
    } catch (error) {
        console.error(error);
    }
}
